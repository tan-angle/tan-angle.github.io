C语言：A+B的几种实现方式

2021-06-04 / 2 min read

代码实现如下：
（1）第一种方式

#include <stdio.h>
int main()
{
    //本方式每次运行只能输入一组数据
    int a, b;
    scanf("%d %d", &a, &b);
    int sum;
    sum = a + b;
    printf("第一种方法sum=%d", sum);
    return 0;
}

（2）第二种方式：有两种
①

#include <stdio.h>
int main()
{
    //本方式每次运行可以输入多组数据
    int a, b, sum = 0;
    //用死循环，在条件满足的时候跳出该循环
    while (1)
    {
        scanf("%d %d", &a, &b);
        if ((a == -1 || b == -1))
            break;
        sum = a + b;
        printf("第二种方法：sum=%d\n", sum);
    }
    return 0;
}

②

#include <stdio.h>
int main()
{
    //本方式每次运行可以输入多组数据
    int a, b;
    int sum;
    while (scanf("%d %d", &a, &b) != EOF)
    {
        if ((a == -1 || b == -1))
            break;
        sum = a + b;
        printf("第二种方法：sum=%d\n", sum);
    }
    return 0;
}

（3）第三种方式

#include <stdio.h>
int main()
{
    //本方式每次运行可以输入多组数据
    int a, b, sum = 0, n;
    printf("您需要计算几组数据，请输入：");
    scanf("%d", &n);
    while (n--)
    {
        scanf("%d %d", &a, &b);
        sum = a + b;
        printf("第三种方法：sum=%d\n", sum);
    }
    return 0;
}
